	\section{Cel projektu}
	Celem projektu jest stworzenie programu w języku C, który służy do automatycznego i estetycznego rozmieszczania
	wierzchołków grafu planarnego w przestrzeni dwuwymiarowej, z wykorzystaniem algorytmu Eadesa lub twierdzenia 
	Tuttego w zależności od wyboru użytkownika. Program działa w trybie nieinteraktywnym/wsadowym.
	\section{Funkcjonalności programu}
	Program oferuje następujące funkcjonalności:
	\begin{enumerate}
		\item Możliwość kontroli sposobu działania programu za pomocą argumentów linii poleceń.
		\item Wczytywanie grafu z pliku tekstowego.
		\item Rozmieszczanie wierzchołków za pomocą algorytmu Eadesa oraz twierdzenia Tuttego.
		\item Zapisywanie informacji o położeniu wierzchołków po ich rozmieszczeniu do pliku tekstowego lub binarnego.
		
	\end{enumerate}
	
	\section{Sposób wykonania programu}
	Schemat wywołania programu wygląda w następujący sposób:
	\begin{mylisting}
./bin/main <plik_wejściowy> <flagi>
	\end{mylisting}
	Na przykład:
	\begin{mylisting}
./bin/main examples/graph_1.txt -t b -o output.bin -s 1234 -a t
	\end{mylisting}
	Powyższa komenda wywoła program w następujący sposób:
	\begin{enumerate}
		\item Program odczyta graf z pliku \texttt{graph\_1.txt}.
		\item Wynik programu zostanie zapisany do pliku binarnego (flaga \texttt{-t}).
		\item Rezultat zostanie zapisany do pliku \texttt{output.bin} (flaga \texttt{-o}).
		\item Program użyje do początkowego rozmieszczenia wierzchołków ziarna \texttt{1234} (flaga \texttt{-s}).
		\item Program użyje twierdzenia Tuttego do rozmieszczenia wierzchołków (flaga \texttt{-a}).
	\end{enumerate}

	\section{Dane wyjściowe}
	Program oferuje możliwość zapisu do pliku tekstowego oraz binarnego. 
	
	\subsection{Plik tekstowy}
	Dane w pliku tekstowym są zapisywane w następującym formacie:
	\begin{mylisting}
<numer_wierzchołka> <współrzędna_x> <współrzędna_y>
	\end{mylisting}
	Na przykład:
	\begin{mylisting}
1 288.000000 288.000000
2 254.000000 254.000000
3 307.000000 307.000000
4 101.000000 101.000000
	\end{mylisting}
	\subsection{Plik binarny}
	Dane w pliku binarnym zapisywane są w następującej kolejności:
	\begin{enumerate}
		\item Liczba wierzchołków - \texttt{uint32\_t} - $4$ B
		\item Identyfikator wierzchołka - \texttt{uint32\_t} - $4$ B
		\item Współrzędna X wierzchołka - \texttt{float} - $4$ B 
		\item Współrzędna Y wierzchołka - \texttt{float} - $4$ B
	\end{enumerate}
	Kroki 2, 3 i 4 powtarzane są dla każdego wierzchołka grafu. Zapis odbywa się zgodnie z metodą \texttt{Little Endian}
	- zapis od najmniej znaczącego bajtu.

	\section{Sposób działania algorytmów}
	\subsection{Algorytm sprężynowy Eadesa}
	Ten algorytm liczy dla każdego wierzchołka sumę sił odpychania i przyciągania.
	\begin{mylisting}
SpringEmbedder(G, minimalna_sila, maks_iteracji)
//minimalna_sila = bardzo mała wartość ogranicza działanie algorytmu i zatrzymuje pracę
//kiedy maksymalna znaleziona siła jest mniejsza niż minimalna_sila
t = 0
max_sila = duza_wartosc // maksymalna siła działająca na jeden z wierzchołków

while (t < maks_iteracji AND max_sila > minimalna_sila)

    dla każdego wierzchołka v w G
        oblicz siły odpychania od wszystkich innych wierzchołków

    dla każdej krawędzi (u, v) w G
        oblicz siłę przyciągania między u i v

    dla każdego wierzchołka v w G
        zaktualizuj pozycję v na podstawie działającej siły
        zaktualizuj max_sila

    t = t + 1

end while
\end{mylisting}
\textbf{Formuła dla siły odpychania:}
\begin{equation}
	f_{\mathrm{rep}}(p_u,p_v) =
	\frac{c_{\mathrm{rep}}}{\|p_v - p_u\|^2} \cdot \vec{p_u p_v}
\end{equation}
Gdzie:
\begin{itemize}
    \item $p_v, p_u$ -- pozycje wierzchołków $v$ i $u$
    \item $c_{\mathrm{rep}}$ -- stała określająca siłę odpychania
    \item $\vec{p_u p_v}$ -- wektor kierunkowy od $u$ do $v$
    \item $\|\cdot\|$ -- norma euklidesowa wektora
\end{itemize}
\textbf{Formuła dla siły przyciągania:}
\begin{equation}
	f_{\mathrm{attr}}(p_u,p_v) =
	c_{\mathrm{spring}} \cdot
	\log\!\left(\frac{\|p_u - p_v\|}{\ell}\right)
	\cdot \vec{p_u p_v}
\end{equation}
Gdzie:
\begin{itemize}
    \item $p_v, p_u$ -- pozycje wierzchołków $v$ i $u$
    \item $c_{\mathrm{spring}}$ -- stała określająca siłę sprężyny
    \item $\ell$ -- długość optymalna krawędzi
    \item $\vec{p_u p_v}$ -- wektor kierunkowy od $u$ do $v$
    \item $\|\cdot\|$ -- norma euklidesowa wektora
\end{itemize}
\subsection{Twierdzenie Tuttego}
	Ten algorytm działa szukając takiej pozycji dla każdego wierzchołka, aby jego współrzędne były średnią arytmetyczną pozycji jego sąsiadów.
	\begin{mylisting}
TutteEmbedding(G)

wybierz wierzchołki zewnętrznej ściany grafu
ustaw ich pozycje na okręgu

dla każdego pozostałego wierzchołka v
    zapisz równanie: pozycja v = średnia pozycji sąsiadów v

zbuduj układ równań liniowych dla współrzędnych x i y

rozwiąż układ równań

dla każdego wierzchołka v
    ustaw jego pozycję według otrzymanego rozwiązania
	\end{mylisting}

\textbf{Warunek barycentryczny:}
\begin{equation}
	p_v = \frac{1}{\deg(v)} \sum_{u \in N(v)} p_u
\end{equation}
Gdzie:
\begin{itemize}
    \item $p_v$ -- pozycja wierzchołka $v$ na płaszczyźnie
    \item $p_u$ -- pozycja sąsiada $u$ wierzchołka $v$
    \item $N(v)$ -- zbiór sąsiadów wierzchołka $v$
    \item $\deg(v)$ -- stopień wierzchołka $v$ (liczba jego sąsiadów)
    \item $\sum$ -- suma po wszystkich sąsiadach wierzchołka $v$
\end{itemize}
\textbf{Równania liniowe dla współrzędnych:}
\begin{equation}
x_v = \frac{1}{\deg(v)} \sum_{u \in N(v)} x_u
\end{equation}
\begin{equation}
y_v = \frac{1}{\deg(v)} \sum_{u \in N(v)} y_u
\end{equation}
Gdzie:
\begin{itemize}
    \item $x_v, y_v$ -- współrzędne wierzchołka $v$
    \item $x_u, y_u$ -- współrzędne sąsiada $u$ wierzchołka $v$
    \item $N(v)$ -- zbiór sąsiadów wierzchołka $v$
    \item $\deg(v)$ -- stopień wierzchołka $v$
\end{itemize}
